\subsection{ The FTC 1 } 

\begin{definition*}
    The FTC 1 states that , if $F$ is differentiable and $\dot F = f$ is continous , then 
    \begin{equation*}
        \int^{b}_{a} f(x) \; dx = F(b) - F(a) = F(x) \Biggr |_{a}^{b} 
    \end{equation*}
\end{definition*}

The FTC1 connects definite integral to Antiderivative . To be consistent with FTC1 
Riemann sum formula for definite integral doesnot need to be changed . 
Consequently , the geometric definition of $\int^{b}_{a} f(x) \; dx$ that is 
consistent with FTC1 is as follows .

\subsection{Properties of Definite Integral}

\begin{enumerate}

     	\item sums
		\begin{equation*}
		    \int^{b}_{a} (f(x)+g(x)) \; dx = \int^{b}_{a} f(x) \; dx + \int^{b}_{a} g(x) \; dx
		\end{equation*}
	\item Constant Multiples
		\begin{equation*}
		    \int^{b}_{a} c \; f(x) \; dx = c \int^{b}_{a} f(x) \; dx
		\end{equation*}
	\item Combining integrals
		\begin{equation*}
		    \int^{c}_{a} f(x)\; dx = \int^{b}_{a} f(x) \; dx + \int^{c}_{b} f(x) dx \text{for any a, b and c}
		\end{equation*}
	\item  
		\begin{equation*}
		    \int^{a}_{a} f(x) \; dx = 0
		\end{equation*}
	

	\item 	if $f(x) \le g(x)$ and $a \le b$ then 
  \[
	\int_{a}^{b} f(x) \; dx \le \int_{a}^{b} g(x) \; dx \text{for} a \le b
  \]
\end{enumerate}

\subsection{Change of Variable for definite integral(Method of substitution)}
  \[
	\int_{u_{1}}^{u_{2}} g(u) \; du = \int_{x_{1}}^{x_{2}} g(u(x)) \; \dot{u}(x) \; dx 
  \]
  where
  \begin{align*}
	  u &= u(x) \\
	  u_{1} &= u(x_{1}) \\
	  u2 &= u(x^{2}) \\
  \end{align*}

CAUTION : Only works if $\dot{u}$ does not change sign 

Example 
\begin{align*}
    &\int_{-1}^{1} x^{2} \; dx \neq 0 \\
	u =& x^{2} \; \implies du = 2x \; dx \implies dx = \frac{1}{2\sqrt{u}} \; du \\
	   &= \int_{1}^{1} u \frac{1}{2\sqrt{u}} \; du \\
	   &\text{thus} \int_{-1}^{1} x^{2} \; dx \neq \int_{1}^{1} \frac{u}{2\sqrt{u}} \; du\\
\end{align*}
 Reason : $\dot{u}$ changes sign , $\dot{u} = 2x > 0 \quad \text{for}\quad x>0 \quad \text{and} \quad  < 0 \quad \text{for}\quad x < 0 $
