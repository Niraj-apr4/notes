\subsection{ The FTC 1 } 

\begin{definition*}
    The FTC 1 states that , if $F$ is differentiable and $\dot F = f$ is continous , then 
    \begin{equation*}
        \int^{b}_{a} f(x) \; dx = F(b) - F(a) = F(x) \Biggr |_{a}^{b} 
    \end{equation*}
\end{definition*}

The FTC1 connects definite integral to Antiderivative . To be consistent with FTC1 
Riemann sum formula for definite integral doesnot need to be changed . 
Consequently , the geometric definition of $\int^{b}_{a} f(x) \; dx$ that is

\[
	\int^{b}_{a} v(t) dt = x(b) -x(a)

	\int^{b}_{a} |{v(t)}| \; dx

\subsection{Properties of Definite Integral}

\begin{enumerate}

     	\item sums
		\begin{equation*}
		    \int^{b}_{a} (f(x)+g(x)) \; dx = \int^{b}_{a} f(x) \; dx + \int^{b}_{a} g(x) \; dx
		\end{equation*}
	\item Constant Multiples
		\begin{equation*}
		    \int^{b}_{a} c \; f(x) \; dx = c \int^{b}_{a} f(x) \; dx
		\end{equation*}
	\item Combining integrals
		\begin{equation*}
		    \int^{c}_{a} f(x)\; dx = \int^{b}_{a} f(x) \; dx + \int^{c}_{b} f(x) dx \text{for any a, b and c}
		\end{equation*}
	\item  
		\begin{equation*}
		    \int^{a}_{a} f(x) \; dx = 0
		\end{equation*}

\end{enumerate}
